* <2023-01-05 Thu> cli directory指定

    def list(*args)
      config = get_config(args).config
      help_dir = options["help_dir"] || config[:local_help_dir]
      puts List.new(help_dir,
                    config[:ext]).list(*args.join(" "))
    end
