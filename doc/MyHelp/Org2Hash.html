def read_item(line)
  m = line.match(/\* (.+)/)
  item = if m
           m[1].match(/head\s*/) ? "head" : m[1]
         else
           nil
         end
  return item
end

def simple_fsm()
  state = :header_read
  item = ""
  @text.split("\n").each do |line|
    next if line.size < 1
    state, action = TRANSITIONS[state][line[0..1]] ||
                    TRANSITIONS[state][:default]
    case action
    when :ignore
    when :start_new_item
      item = read_item(line)
      @contents[item] = ""
    when :add_contents
      @contents[item] += line + "\n"
    end
  end
end
