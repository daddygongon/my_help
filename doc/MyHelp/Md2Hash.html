def fsm
  contents = {}
  state = :init
  @results = []
  item = ""
  @md_text.each do |line|
    state, action = TRANS[state][line[0]] || TRANS[state][:default]

    if line[0..1] == "#+" # line[1] != " "
      state = :init
      action = :ignore
    end

    results << [line, state, action]

    case action
    when :ignore
    when :item
      item = line.match(/^# (.+)/)[1]
      contents[item] = []
    when :data
      contents[item] << line
    end
  end
  return contents
end
