def cp_templates
  target_dir = @config[:local_help_dir]
  src_dir = @config[:template_dir]
  ext = @config[:ext]
  Dir.glob(File.join(src_dir, "*#{ext}")).each do |file|
    FileUtils.cp(file, target_dir, verbose: false)
  end
end
